Cross Validation

library(caret)
## Loading required package: lattice
## Loading required package: ggplot2
numFolds = trainControl(method="cv",number=10)
cpGrid = expand.grid(.cp = seq(0.01,0.5,0.01))

set.seed(111)
cvmodels = train(Life.Exp~., 
      data=statedata,
      method="rpart",
      trControl=numFolds,
      tuneGrid=cpGrid)
